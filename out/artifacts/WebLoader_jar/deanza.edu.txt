			var imageCache = [];
			
			function pausePlaySlide() {
				if(Cap) {clearTimeout(Cap);}
				clearInterval(SlideShow);
				if( paused == false ) {
					paused = true;       
					bckGrnImg = document.getElementById("pauseButton");
					bckGrnImg.setAttribute("class", "play");    
				} else {
					paused = false;     
					bckGrnImg = document.getElementById("pauseButton");
					bckGrnImg.setAttribute("class", "pause");
					SlideShow = setInterval("RunSlideShow('primaryImage','img_div')", 3000 ); // restart up slideshow, 3 sec delay
				}
			}
	
			function revSlide() {
				if(paused == false) { 
					pausePlaySlide();        
					currentImage--; 
				}
				currentImage--; 
				if( currentImage < 0 ) { 
					currentImage = slideCache.length - 1;
				}
				if(Cap) {clearTimeout(Cap);}
				clearInterval(SlideShow);
				FadeInImage("primaryImage",slideCache[currentImage],"img_div");
			}
	
			function advSlide() {
				if(paused == false) { 
					pausePlaySlide();       
				} 
				currentImage++; 
				if( currentImage > slideCache.length - 1 ) { 
					currentImage = 0;
				}
				if(Cap) {clearTimeout(Cap);}
				clearInterval(SlideShow);
				FadeInImage("primaryImage",slideCache[currentImage],"img_div");
			}
	
			function loadSlideImages() {
				for( i = 1; i < slideCache.length; i++) {

				}
				SlideShow = setInterval("RunSlideShow('primaryImage','img_div')", 5000 ); // start slide show 5 sec delay

	// Browser Slide-Show script. With image cross fade effect for those browsers that support it.
	// Script copyright (C) 2004-2010 www.cryer.co.uk.
	// Script is free to use provided this copyright header is included.
	// modifications by De Anza Web Team
	
	var FadeDurationMS = 1000;
	var Cap = false;
	var SlideShow = 0;
	var paused = false;
	var hasMSIE = false	
	function setCaption()
	{
		var domObjectCap = document.getElementById("slidecaption");
		
		if(!paused)
		{
			if( currentImage == 0)
			{ currentCaption = slideCache.length - 1; }
			else
			{ currentCaption = currentImage - 1; }
		}
		else
		{ currentCaption = currentImage; }
		
		domObjectCap.innerHTML = captionCache[currentCaption];
	}
	
	function SetOpacity(object,opacityPct)
	{
		if(hasMSIE){
			object.style.filter = 'alpha(opacity=' + opacityPct + ')';
		}
		object.style.MozOpacity = opacityPct/100;
		object.style.opacity = opacityPct/100;
	}
	
	function ChangeOpacity(id,msDuration,msStart,fromO,toO)
	{
		var element = document.getElementById(id);
		var msNow = (new Date()).getTime();
		
		var opacity = fromO + (toO - fromO) * (msNow - msStart) / msDuration;
		
		if (opacity >= 100)
		{
			SetOpacity(element,100);
			element.timer = undefined;        
		}
		else if (opacity <= 0)
		{
			SetOpacity(element,0);
			element.timer = undefined;
		}
		else 
		{
			SetOpacity(element,opacity);
			element.timer = window.setTimeout("ChangeOpacity('" + id + "'," + msDuration + "," + msStart + "," + fromO + "," + toO + ")",10);
		}
	}
	
	function FadeInImage(foregroundID,newImage,backgroundID,imageAlt)
	{
		var foreground = document.getElementById(foregroundID);
	
		if (foreground.timer) window.clearTimeout(foreground.timer);
	
		if (backgroundID)
		{
			var background = document.getElementById(backgroundID);
			
			if (background)
			{
				if (background.src)
				{
					foreground.src = background.src; 
					SetOpacity(foreground,100);
				} 
				
				Cap = setTimeout("setCaption()", 500);      
	
				background.src = newImage;
				background.style.backgroundImage = 'url(' + newImage + ')';
				background.style.backgroundRepeat = 'no-repeat';
				var startMS = (new Date()).getTime();
				foreground.timer = window.setTimeout("ChangeOpacity('" + foregroundID + "'," + FadeDurationMS + "," + startMS + ",100,0)",10);
			}
		} 
		else 
		{
			foreground.src = newImage;
		}
		foreground.alt = imageAlt;
	}
	
	function RunSlideShow(pictureID,backgroundID)
	{   
		if (slideCache[currentImage])
		{     
			if(! paused)
			{ 
				FadeInImage(pictureID,slideCache[currentImage],backgroundID,altCache[currentImage]);
				
				if( currentImage == (slideCache.length - 1) )
				{ 
					currentImage = 0;
				}
				else
				{
					currentImage++;
				}
			}
		}
	}
